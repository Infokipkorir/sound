// ─── WORKER SLIDER PAGINATION ─────────────────────────────────────────────────
const PAGE_SIZE = 6;

function buildPages(arr, size){
  const pages=[];
  for(let i=0;i<arr.length;i+=size) pages.push(arr.slice(i,i+size));
  return pages;
}

function renderWorkerPage(){
  if(!workerPages.length) return;
  const page = workerPages[workerPage] || [];
  renderWorkersFeed(page);
  updateDots('workers-dots', workerPages.length, workerPage);

  document.getElementById('workers-next').disabled = workerPage>=workerPages.length-1;
}
function slideWorkers(dir){

}

  updateDots('jobs-dots', jobPages.length, jobPage);

  renderJobPage();
}

  ).join('');
}

// ─── MAIN DATA LOAD ───────────────────────────────────────────────────────────
async function loadAllData() {

    sb.from('worker_profiles')
      .select('id, full_name, profile_photo_url, county, preferred_role, about_text, key_skills, salary_range, salary_min, salary_max, availability, edu_level, is_admin_managed, submitted_at, view_count')
      .eq('status', 'approved')
      .eq('is_deleted', false)
      .order('submitted_at', { ascending: false })
      .limit(30),
    sb.from('job_postings')
      .select('id, title, description, location, county, salary_min, salary_max, salary_range, work_type, job_role, is_urgent, is_admin_post, created_at, view_count, employer_profiles(full_name, company_name, logo_url, is_verified)')
      .eq('status', 'active')
      .order('created_at', { ascending: false })
      .limit(30),
    sb.from('testimonials').select('id, user_name, message, rating').eq('approved', true).order('created_at', { ascending: false }).limit(6),
    sb.from('worker_profiles').select('id', { count: 'exact', head: true }).eq('status','approved').eq('is_deleted',false),
    sb.from('employer_profiles').select('id', { count: 'exact', head: true }),
    sb.from('job_postings').select('id', { count: 'exact', head: true }).eq('status','active'),
  ]);

  // Stats
  const tw = totalWRes.count || 0, te = totalERes.count || 0, tj = totalJRes.count || 0;
  document.getElementById('hero-badge-count').textContent = tw > 0 ? tw.toLocaleString()+'+' : 'Growing';
  const swEl=document.getElementById('stat-workers'), seEl=document.getElementById('stat-employers'), sjEl=document.getElementById('stat-jobs');
  if(tw>0){swEl.dataset.suffix='+';animateCount(swEl,tw);}else swEl.textContent='Growing';
  if(te>0){seEl.dataset.suffix='+';animateCount(seEl,te);}else seEl.textContent='Growing';
  if(tj>0){sjEl.dataset.suffix='+';animateCount(sjEl,tj);}else sjEl.textContent='Growing';

  allWorkers = wRes.data || [];
  allJobs    = jRes.data || [];

  // Update tab count badges
  const tnW = document.getElementById('tab-workers-count');
  const tnJ = document.getElementById('tab-jobs-count');
  if(tnW) tnW.textContent = allWorkers.length > 0 ? allWorkers.length : '';
  if(tnJ) tnJ.textContent = allJobs.length > 0 ? allJobs.length : '';

  // Build pages & render
  workerPages = buildPages(allWorkers, PAGE_SIZE);
  jobPages    = buildPages(allJobs, PAGE_SIZE);

  if(workerPages.length > 1) document.getElementById('workers-nav').style.display='flex';
  if(jobPages.length > 1)    document.getElementById('jobs-nav').style.display='flex';

  renderWorkerPage();
  renderJobPage();

function renderTestimonials(list) {
  const el = document.getElementById('testi-grid');
  if (!list.length) {
    el.innerHTML = `<div class="testi-empty">No testimonials yet — be the first to share your experience!</div>`;
    return;
  }
  const stars = r => '<i class="bi bi-star-fill" style="color:#F59E0B;font-size:12px"></i>'.repeat(Math.min(r||5,5));
  el.innerHTML = list.map(t => `
    <blockquote class="testi-card">
      <p>"${escH(t.message)}"</p>
      <div class="testi-author">
        <div class="tavi">${initials(t.user_name)}</div>
        <div><strong>${escH(t.user_name)}</strong><br><span style="color:rgba(255,255,255,.6);font-size:12px">${stars(t.rating)}</span></div>
      </div>
    </blockquote>`).join('');
}

async function submitPrivacyRequest() {
  const btn = document.getElementById('pf-submit-btn');
  const res = document.getElementById('pf-result');
  const name     = document.getElementById('pf-name').value.trim();
  const phone    = document.getElementById('pf-phone').value.trim();
  const email    = document.getElementById('pf-email').value.trim();
  const location = document.getElementById('pf-location').value.trim();
  const details  = document.getElementById('pf-details').value.trim();
  if (!name || !phone || !location) {
    res.style.display='block'; res.style.background='#FCEBEB'; res.style.color='#A32D2D';
    res.textContent='Please fill in Name, Phone and Location.';
    return;
  }
  let roleField, salaryField, extraData = {};
  if (currentTab === 'worker') {
    roleField   = document.getElementById('pf-role-worker').value;
    salaryField = document.getElementById('pf-salary-worker').value;
    extraData   = { experience: document.getElementById('pf-exp').value, availability: document.getElementById('pf-avail').value };
  } else {
    roleField   = document.getElementById('pf-role-employer').value;
    salaryField = document.getElementById('pf-budget').value;
    extraData   = { work_type: document.getElementById('pf-work-type').value, workers_needed: document.getElementById('pf-count').value };
  }
  btn.disabled = true; btn.textContent = 'Submitting…';
  const { error } = await sb.from('privacy_post_requests').insert({
    requester_type: currentTab, full_name: name, phone, email: email||null, location,
    job_role: roleField||null, salary_info: salaryField||null, extra_data: extraData, details: details||null,
    status: 'pending', created_at: new Date().toISOString(),
  });
  btn.disabled = false; btn.innerHTML = '<i class="bi bi-send-fill me-1"></i> Submit to Admin';
  if (error) {
    res.style.display='block'; res.style.background='#FCEBEB'; res.style.color='#A32D2D';
    res.textContent = 'Submission failed: '+error.message; return;
  }
  res.style.display='block'; res.style.background='var(--green-light)'; res.style.color='var(--green-dark)';
  res.innerHTML = '<i class="bi bi-check-circle-fill me-1"></i><strong>Request received!</strong> Our admin will review and post your listing within 24 hours. We\'ll reach you on '+phone+'.';
  ['pf-name','pf-phone','pf-email','pf-location','pf-details','pf-salary-worker','pf-budget'].forEach(id=>{ const el=document.getElementById(id); if(el) el.value=''; });
}
